def derive_key(shared_secret: bytes, salt: bytes | None = None) -> bytes:
    """Derive encryption key from shared secret - FIXED VERSION"""
    try:
        if salt is None:
            salt = config['salt'].encode()
        
        # Ensure shared_secret is exactly 32 bytes
        if len(shared_secret) < 32:
            shared_secret = shared_secret.ljust(32, b'\x00')
        elif len(shared_secret) > 32:
            shared_secret = shared_secret[:32]
        
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,  # AES-256 requires 32 bytes
            salt=salt,
            iterations=config['iterations'],
            backend=backend
        )
        derived = kdf.derive(shared_secret)
        
        # CRITICAL: Validate derived key
        if len(derived) != 32:
            raise ValueError(f"Key derivation failed: got {len(derived)} bytes, expected 32")
        
        return derived
    except Exception as e:
        raise Exception(f"Key derivation failed: {str(e)}")

def ratchet_key(current_key: bytes, version: int | None = None) -> bytes:
    """
    Deterministic key ratcheting - FIXED VERSION
    """
    if version is None:
        version = 1
    
    try:
        # Validate input
        if len(current_key) != 32:
            raise ValueError(f"Invalid key length: {len(current_key)}, expected 32")
        
        # Use SHA256 for deterministic ratcheting
        h = hashlib.sha256()
        h.update(b'ratchet_v1')  # Domain separator
        h.update(current_key)
        h.update(str(version).encode())
        new_key = h.digest()
        
        # Validate output
        if len(new_key) != 32:
            raise ValueError(f"Ratchet produced invalid key length: {len(new_key)}")
        
        return new_key
    except Exception as e:
        raise Exception(f"Key ratcheting failed: {str(e)}")

def encrypt_message(message: bytes | str, key: bytes) -> bytes:
    """
    AES-256-CBC encryption - BULLETPROOF VERSION
    Format: IV (16 bytes) + Ciphertext (padded)
    """
    try:
        # Validate inputs
        if isinstance(message, str):
            message = message.encode('utf-8')
        
        if len(key) != 32:
            raise ValueError(f"Invalid key length: {len(key)} bytes, expected 32")
        
        if len(message) == 0:
            raise ValueError("Cannot encrypt empty message")
        
        # Generate random IV
        iv = os.urandom(16)
        
        # Create cipher
        cipher = Cipher(algorithms.AES(key), modes.CBC(iv), backend=backend)
        encryptor = cipher.encryptor()
        
        # Apply PKCS7 padding
        padder = crypto_padding.PKCS7(128).padder()
        padded = padder.update(message) + padder.finalize()
        
        # Encrypt
        ciphertext = encryptor.update(padded) + encryptor.finalize()
        
        # Combine IV + ciphertext
        result = iv + ciphertext
        
        # Validate output
        if len(result) < 32:  # IV(16) + at least 1 block(16)
            raise ValueError(f"Encryption produced invalid output: {len(result)} bytes")
        
        return result
        
    except Exception as e:
        raise Exception(f"Encryption failed: {str(e)}")

def decrypt_message(encrypted_data: bytes, key: bytes) -> bytes:
    """
    AES-256-CBC decryption - BULLETPROOF VERSION
    """
    try:
        # Validate inputs
        if len(encrypted_data) < 32:  # IV(16) + minimum ciphertext(16)
            raise ValueError(f"Encrypted data too short: {len(encrypted_data)} bytes")
        
        if len(key) != 32:
            raise ValueError(f"Invalid key length: {len(key)} bytes, expected 32")
        
        # Extract IV and ciphertext
        iv = encrypted_data[:16]
        ciphertext = encrypted_data[16:]
        
        # Validate ciphertext is block-aligned
        if len(ciphertext) % 16 != 0:
            raise ValueError(f"Ciphertext not block-aligned: {len(ciphertext)} bytes")
        
        # Create cipher
        cipher = Cipher(algorithms.AES(key), modes.CBC(iv), backend=backend)
        decryptor = cipher.decryptor()
        
        # Decrypt
        decrypted_padded = decryptor.update(ciphertext) + decryptor.finalize()
        
        # Remove PKCS7 padding
        unpadder = crypto_padding.PKCS7(128).unpadder()
        decrypted = unpadder.update(decrypted_padded) + unpadder.finalize()
        
        # Validate output
        if len(decrypted) == 0:
            raise ValueError("Decryption resulted in empty plaintext")
        
        return decrypted
        
    except Exception as e:
        raise Exception(f"Decryption failed: {str(e)}")
